name: Task CI Pipeline

  repository_dispatch:
    types: [run-task]

  ci:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4

          node-version: 20

      - name: Install dependencies
        run: npm install

      - name: Lint
        run: npm run lint

      - name: Typecheck
        run: npm run typecheck

      - name: Test
        run: npm run test

      - name: Run DB Migration
        run: npm run db:migrate

      - name: Build
        run: npm run build

      - name: Deploy (staging)
        run: npm run deploy:stg
        env:
          SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
          SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}

          SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
          TASK_ID: ${{ github.event.client_payload.task_id }}

📁 2. scripts/trigger-ci.ts
ts
export async function triggerCI(task: {
  task_id: string;
  run_number: number;
  env?: "dev" | "stg" | "prd";
}) {
  await fetch("https://api.github.com/repos/YOUR_ORG/YOUR_REPO/dispatches", {
    method: "POST",
    headers: {
      Authorization: `token ${process.env.GITHUB_TOKEN}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      event_type: "run-task",
      client_payload: {
        task_id: task.task_id,
        run_number: task.run_number,
        env: task.env ?? "stg",
      },
    }),
  });
}

📁 3. scripts/update-task-status.js
js
import { createClient } from "@supabase/supabase-js";

const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_SERVICE_ROLE_KEY
);

async function update() {
  const taskId = process.env.TASK_ID;

  if (!taskId) throw new Error("TASK_ID missing");

  const { error } = await supabase
    .from("task_pipeline_logs")
    .update({
      status: "DEPLOYED",
      finished_at: new Date().toISOString(),
    })
    .eq("task_id", taskId);

  if (error) {
    console.error(error);
    process.exit(1);
  }

  console.log("Task updated:", taskId);
}

update();

📁 5. Qwik API (trigger CI from UI)

// src/routes/api/run-task/index.ts
import { server$ } from "@builder.io/qwik-city";

export const onPost = server$(async ({ request }) => {
  const body = await request.json();

  await fetch("https://api.github.com/repos/YOUR_ORG/YOUR_REPO/dispatches", {
    method: "POST",
    headers: {
      Authorization: `token ${process.env.GITHUB_TOKEN}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      event_type: "run-task",
      client_payload: {
        task_id: body.task_id,
        run_number: body.run_number,
        env: body.env || "stg",
      },
    }),
  });

  return { ok: true };
});

📁 6. Simple UI Button (Qwik)

import { component$ } from "@builder.io/qwik";

export default component$(() => {
  const runTask = async () => {
    await fetch("/api/run-task", {
      method: "POST",
      body: JSON.stringify({
        task_id: "TASK-DB-001",
        run_number: 1,
        env: "stg",
      }),
    });
  };

  return (
    <button onClick$={runTask}>
      🚀 Run CI Task
    </button>
  );
